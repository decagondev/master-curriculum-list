Linked List Operations

Learn how to implement insertion and deletion operations on linked lists, fundamental operations for managing dynamic data structures.

Linked List Fundamentals

Node Structure

class Node {
  constructor(data) {
    this.data = data;
    this.next = null;
  }
}

Insertion Operations

There are three main types of insertions in a singly linked list:

1. Insert at the Beginning

function insertAtBeginning(head, data) {
  // Create a new node
  const newNode = new Node(data);
  
  // Point new node to current head
  newNode.next = head;
  
  // Update head to new node
  return newNode;
}

2. Insert at the End

function insertAtEnd(head, data) {
  // Create a new node
  const newNode = new Node(data);
  
  // If list is empty, make new node the head
  if (!head) {
    return newNode;
  }
  
  // Traverse to last node
  let current = head;
  while (current.next) {
    current = current.next;
  }
  
  // Link last node to new node
  current.next = newNode;
  
  return head;
}

3. Insert at a Specific Position

function insertAtPosition(head, data, position) {
  // Create a new node
  const newNode = new Node(data);
  
  // If position is 0, insert at beginning
  if (position === 0) {
    newNode.next = head;
    return newNode;
  }
  
  // Traverse to position - 1
  let current = head;
  let count = 0;
  
  while (current && count < position - 1) {
    current = current.next;
    count++;
  }
  
  // If position is invalid
  if (!current) {
    console.log("Position out of bounds");
    return head;
  }
  
  // Insert node at position
  newNode.next = current.next;
  current.next = newNode;
  
  return head;
}

Deletion Operations

There are two main approaches to deletion in a singly linked list:

1. Delete by Position

function deleteAtPosition(head, position) {
  // If list is empty
  if (!head) {
    return null;
  }
  
  // If deleting the head
  if (position === 0) {
    return head.next;
  }
  
  // Find the node before the one to delete
  let current = head;
  let count = 0;
  
  while (current.next && count < position - 1) {
    current = current.next;
    count++;
  }
  
  // If position is invalid
  if (!current.next) {
    console.log("Position out of bounds");
    return head;
  }
  
  // Skip the node to delete
  current.next = current.next.next;
  
  return head;
}

2. Delete by Value

function deleteByValue(head, value) {
  // If list is empty
  if (!head) {
    return null;
  }
  
  // If head has the value to delete
  if (head.data === value) {
    return head.next;
  }
  
  // Search for the node with the value
  let current = head;
  
  while (current.next && current.next.data !== value) {
    current = current.next;
  }
  
  // If value not found
  if (!current.next) {
    console.log("Value not found");
    return head;
  }
  
  // Skip the node with the value
  current.next = current.next.next;
  
  return head;
}

Testing Your Implementation

// Create a linked list
let head = new Node(1);
head.next = new Node(2);
head.next.next = new Node(3);

// Print linked list
function printList(head) {
  let current = head;
  const values = [];
  
  while (current) {
    values.push(current.data);
    current = current.next;
  }
  
  console.log(values.join(" -> "));
}

console.log("Original list:");
printList(head);  // Output: 1 -> 2 -> 3

// Insert 0 at the beginning
head = insertAtBeginning(head, 0);
console.log("After inserting 0 at beginning:");
printList(head);  // Output: 0 -> 1 -> 2 -> 3

// Insert 4 at the end
head = insertAtEnd(head, 4);
console.log("After inserting 4 at end:");
printList(head);  // Output: 0 -> 1 -> 2 -> 3 -> 4

// Insert 2.5 at position 4 (after the 3)
head = insertAtPosition(head, 2.5, 4);
console.log("After inserting 2.5 at position 4:");
printList(head);  // Output: 0 -> 1 -> 2 -> 3 -> 2.5 -> 4

// Delete node at position 1
head = deleteAtPosition(head, 1);
console.log("After deleting node at position 1:");
printList(head);  // Output: 0 -> 2 -> 3 -> 2.5 -> 4

// Delete node with value 2.5
head = deleteByValue(head, 2.5);
console.log("After deleting node with value 2.5:");
printList(head);  // Output: 0 -> 2 -> 3 -> 4

Edge Cases to Consider

  • Empty List: Always check if the head is null before operations
  • Single Node: Be careful when deleting the only node in the list
  • First/Last Node: Operations on boundary nodes require special handling
  • Invalid Position: Check if the position is within bounds
  • Node Not Found: Handle cases where a value being searched doesn't exist
