Implementation Examples

Recursive Search

public class BinaryTree {
    private BinaryTreeNode root;

    public BinaryTreeNode search(T value) {
        return searchRecursive(root, value);
    }

    private BinaryTreeNode searchRecursive(BinaryTreeNode node, T value) {
        // Base cases
        if (node == null) return null;
        if (node.getValue().equals(value)) return node;

        // Recursive cases
        BinaryTreeNode left = searchRecursive(node.getLeft(), value);
        if (left != null) return left;

        return searchRecursive(node.getRight(), value);
    }
}

Level-First Traversal

public List levelOrderTraversal() {
    if (root == null) return new ArrayList<>();
    
    List result = new ArrayList<>();
    Queue> queue = new LinkedList<>();
    queue.offer(root);
    
    while (!queue.isEmpty()) {
        BinaryTreeNode node = queue.poll();
        result.add(node.getValue());
        
        if (node.getLeft() != null) queue.offer(node.getLeft());
        if (node.getRight() != null) queue.offer(node.getRight());
    }
    
    return result;
}

Depth-First Traversal

public class DepthFirstTraversal {
    // Recursive implementation
    public List inorderTraversal(BinaryTreeNode root) {
        List result = new ArrayList<>();
        inorderHelper(root, result);
        return result;
    }

    private void inorderHelper(BinaryTreeNode node, List result) {
        if (node != null) {
            inorderHelper(node.getLeft(), result);
            result.add(node.getValue());
            inorderHelper(node.getRight(), result);
        }
    }

    // Iterative implementation using Stack
    public List inorderTraversalIterative(BinaryTreeNode root) {
        List result = new ArrayList<>();
        Stack> stack = new Stack<>();
        BinaryTreeNode current = root;

        while (current != null || !stack.isEmpty()) {
            while (current != null) {
                stack.push(current);
                current = current.getLeft();
            }
            current = stack.pop();
            result.add(current.getValue());
            current = current.getRight();
        }
        return result;
    }
}
